## Examples

### Unordered List
```js
import ListItem from './Components/ListItem';

<List>
  <ListItem>Cartoon Network</ListItem>
  <ListItem>CNN</ListItem>
  <ListItem>TBS</ListItem>
  <ListItem>TNT</ListItem>
</List>
```

### Ordered List
```js
import ListItem from './Components/ListItem';

<List ordered>
  <ListItem>Cartoon Network</ListItem>
  <ListItem>CNN</ListItem>
  <ListItem>TBS</ListItem>
  <ListItem>TNT</ListItem>
</List>
```

### Size
```js
import ListItem from './Components/ListItem';

<List size="3">
  <ListItem>Cartoon Network</ListItem>
  <ListItem>CNN</ListItem>
  <ListItem>TBS</ListItem>
```
